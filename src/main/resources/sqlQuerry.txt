CREATE DATABASE Eventine;

USE Eventine;

CREATE TABLE event(
id integer primary key auto_increment not null,
event_name varchar(60) not null,
description varchar (500),
owner_id integer not null,
participants_id integer,
FOREIGN KEY (owner_id) REFERENCES user(id),
FOREIGN KEY (participants_id) REFERENCES user(id)
);

CREATE TABLE user(
id  integer primary key auto_increment not null,
email varchar(40) not null,
user_name varchar(60),
user_role varchar(10),
participing_events_id integer,
owned_events_id integer,
FOREIGN KEY (participing_events_id) REFERENCES event(id),
FOREIGN KEY (owned_events_id) REFERENCES event(id)
);
